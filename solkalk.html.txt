    <script>
        async function convertSolarTime() {
            const city = document.getElementById("city").value;
            if (!city) {
                document.getElementById("solar_result").innerText = "Пожалуйста, введите город.";
                return;
            }
            const response = await fetch(`https://worldtimeapi.org/api/timezone/Europe/Moscow`);
            const data = await response.json();
            document.getElementById("solar_result").innerText = `Местное время в ${city}: ${data.datetime}`;
        }
        
        function calculateBaZi() {
            const birthDate = document.getElementById("birth_date").value;
            if (!birthDate) {
                document.getElementById("bazi_result").innerText = "Пожалуйста, введите дату рождения.";
                return;
            }
            document.getElementById("bazi_result").innerText = `Рассчет Ба Цзы для ${birthDate}: Пример данных.`;
        }
    </script>
